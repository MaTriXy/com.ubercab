function computeRoutes(
  riders, route, used, n) {
  if (n == riders.length) {
    saveRoute(route);
    return;
  }
  for (var i = 0;
       i < riders.length;
       i++) {
     if (used[i]) continue;
    route.push(
      riders[i].location);
    used[i] = true;
    computeRoutes(
      riders,
      route,
      used,
      n + 1);
    used[i] = false;
    route.pop();
  }
}
