def sort_fares(a, low, high):

  if low < high:

    p = partition(a, low, high)

    sort_fares(a, low, p - 1)

    sort_fares(a, low, high)


def partition(a, low, high):

  pivot = a[high]

  p = low

  for i in xrange(low, high):

    if a[i] <= pivot:

      a[p], a[i] = a[i], a[p]

      p += 1

  a[p], a[high] = a[high], a[p]

  return p
